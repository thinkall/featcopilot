üè™ Native Feature Store Integration


from featcopilot.stores import FabricFeatureStore

# Save generated features to Fabric Feature Store
store = FabricFeatureStore(
    workspace='data-science-workspace',
    entity_columns=['customer_id'],
    auto_materialize=True
)

store.save_features(
    df=df_features,
    feature_view_name='customer_churn_features',
    description='Auto-generated features for churn prediction'
)

# Retrieve for real-time inference
online_features = store.get_online_features(
    entity_dict={'customer_id': [101, 102, 103]}
)
